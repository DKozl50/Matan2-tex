	$\displaystyle \sum\limits_{n = 1}^{\infty} \frac{cos \; \sqrt{2}n}{2n - 5}$ \\

		& \text{Тогда } \; 
		\left( \sum_{n=0}^{\infty} \dfrac{(-1)^n}{(2n + 1)!} \right) \cdot \left( \sum_{n=0}^{\infty} \dfrac{(-1)^n}{(2n)!} \right) =

		\sum_{n = 0}^{\infty} \dfrac{(-1)^{n} \cdot 4^n}{(2n + 1)!}
	\end{flalign*}
